```Pseudo
inclusionCoefficient(array A, array B) -> float ic
cardIntersezione = 0
mergeSort(B)
for i = 0 to n
	if binarySearch(B, A[i]) != -1
		cardIntersezione ++
ic = cardIntersezione / n
return ic
```
